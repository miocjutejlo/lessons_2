# привязать git к репозиторию
git init (Создается .git папка, где вся история будет)

# отвязать git
rm -rf .git (Удаилть .git)

# посмотреть текущие изменения
git status

# добавить файл для коммита
git add путь_до_файла
git add . (все измененные файлы)

# убрать файл из будущего коммита
git rm --cached путь_до_файла
git rm -r --cached путь_до_папки (убрать из папки все файлы)

# сделать коммит
git commit -m "сообщение_коммита"

# связать локальный проект к github

git remote add origin ссылка_на_репозиторий

# посмотреть все связи с репозиториями

git remote -v

# вылить изменения в репозиторий

git push
git push --set-upstream origin название_ветки (Если ветки нет в репозитории)

# получить изменения из репозитория

git pull
